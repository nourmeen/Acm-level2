#include<iostream>
#include <stdio.h>
#include<cmath>
#include<string>
#include <algorithm>
#include<math.h>
#include<map>
#include<set>
#include<vector>
using namespace std;
int main(){
	std::ios_base::sync_with_stdio(0);
	cin.tie(NULL); cout.tie(0);
	set<string>str;
	bool bol = 0;
	string s1, s2;
	while (cin >> s1){
		s2 = "";
		int x = s1.length();
		for (int i = 0; i < x; i++){
			if (s1[i] == '-')
				bol = 1;
			else
				s2 += s1[i];
			if (s1[i] <= 'Z'&&s1[i] >= 'A'){
				s2 += (s1[i] + 32);
			}
			else
			if (s1[i] >= 'a'&&s1[i] <= 'z'){
				s2 += s1[i];
			}
			else
			{
				str.insert(s2);
				s2 = "";
			}
		}
		if (!bol){
			str.insert(s2);
			s2 = "";
		}
	}
	set<string>::iterator it;
	for (it = str.begin(); it != str.end(); it++){
		if (*it != "")
			cout << *it << "\n";
	}
	return 0;

}
